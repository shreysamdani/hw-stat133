``` r
# Problem 1

fairDie = die()
sixes = 0
for (i in 1:1000) {
  trial = roll(fairDie, 4)
  if (is.element(6, trial$rolls)) {
    sixes = sixes + 1
  } 
}
sixes/1000
```

``` r
# Problem 2
sixes = 0
for (i in 1:1000) {
  for (j in 1:24) {
    trial = roll(fairDie, 2)
    if (sum(trial$rolls == 6) == 2) {
      sixes = sixes + 1
      break
    }
  }
}
sixes / 1000
```
